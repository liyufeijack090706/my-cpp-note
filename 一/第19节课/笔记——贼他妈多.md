~~~c++
	//break————————停止当前循环
	for (int i = 1; i <= 100; i++) {
		if (i == 50) {
			cout << i << endl;
			break;
		}
		cout << i << endl;
	}
	//continue————————跳过本次循环，进入下一次循环
	for (int i = 1; i <= 100; i++) {
		if (i == 50) {
			cout << i << endl;
			continue;
		}
		cout << i << endl;
	}
~~~
